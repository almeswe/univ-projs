script-line-argument ::= argument ':' value { ',' value }


shape    ::= 'ellipse' | 'circle' | 'dot' | 'polygon' | 'line' | 'triangle' | 'rect' | 'square'
value    ::= [0-9]([0-9]*)
argument ::= [a-zA-Z]([a-zA-Z0-9]*)
